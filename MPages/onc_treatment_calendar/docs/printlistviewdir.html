    This module is useful to construct the Page 2 Print preview of Calendar with appointments,prescriptions and notes information along with date and time.

    angular.module('calview.directives').directive('printListView', [
  '$timeout', function($timeout) {
    return {
      templateUrl: "printlistviewdir.html",
      replace: true,
      restrict: 'E',
      link: function(scope, iElement, iAttrs, controller) {
        var populateFirstColumn, pushSchedListAccrossColumns, watchers;
        scope.startingSlice = [];
        scope.maxColHeight = 580;
        scope.currentColHeight = 0;
        scope.schedCount = 0;
        scope.$on('printSliceChanged', function(ev, slice) {
          scope.startingSlice = slice;
          iElement.children().empty();
          populateFirstColumn();
        });
        
        /*scope.$on('updatePaperSize', function(ev, slice) {
         pushSchedListAccrossColumns();
        });*/
        watchers = void 0;
        scope.$on('suspendWatchers', function() {
          watchers = scope.$$watchers;
          scope.$$watchers = [];
        });
        scope.$on("resumeWatchers", function() {
          if (watchers) {
            scope.$$watchers = watchers;
          }
          watchers = void 0;
        });
        populateFirstColumn = function() {
            var date, dateLineDiv, momentDate, parentDiv, schedType, schedulableDiv, _i, _len, _ref, tempFlag ;
          _ref = scope.startingSlice;

          for (_i = 0, _len = _ref.length; _i < _len; _i++) {
            date = _ref[_i];
            if (date.schedulables.length > 0) {
              scope.schedCount = 0;
              scope.dateflag = false;
              while (scope.schedCount < date.schedulables.length) {
              	
              	schedType = date.schedulables[scope.schedCount].schedType;
              	
              	if(date.schedulables[scope.schedCount].visible === true){
	              	if(scope.dateflag  === false) {
	              	  momentDate = moment(date.unixStamp);
		              dateLineDiv = '<div><strong>' + momentDate.format('MMMM D,YYYY dddd') + '</strong></div>';
		              iElement.children(':first').append(dateLineDiv);
		              scope.dateflag = true;
					}
	                parentDiv = iElement.children(':first').children(':last');
	           
	                if (schedType === 'MedInstance') {
	                schedulableDiv = '<div><strong><i>&#160;&#160;' + date.schedulables[scope.schedCount].time() 
					+ " " + date.schedulables[scope.schedCount].timeZone() +
					'</i></strong>&#160;&#160;' + date.schedulables[scope.schedCount].name() +', '+  date.schedulables[scope.schedCount].dose() + '</div>';
	                } else if (schedType === 'NoteInstance') {
	                schedulableDiv = '<div><strong><i>&#160;&#160;' + date.schedulables[scope.schedCount].time() +
					'</i></strong>&#160;&#160;' + date.schedulables[scope.schedCount].name() + '</div>';
	                }else {
	                schedulableDiv = '<div><strong><i>&#160;&#160;' + date.schedulables[scope.schedCount].time() 
					+ " " + date.schedulables[scope.schedCount].timeZone() +
					'</i></strong>&#160;&#160;' + date.schedulables[scope.schedCount].name() + '</div>';
	                } 
	                parentDiv.append(schedulableDiv);
	                
	              }
	              scope.schedCount = scope.schedCount + 1;
              }
            }
          }
          $timeout((function() {
            pushSchedListAccrossColumns();
          }), 1);
        };
        /* There is no way to know how tall the divs will be until after they are rendered.  Therefore they must be rendered in one
        column first and then pushed to the other columns based on height.  */
        pushSchedListAccrossColumns = function() {
          var divHolder;
          if (iElement.children(':first').height() > scope.maxColHeight) {
            while (iElement.children(':first').height() > scope.maxColHeight) {
              log.debug(iElement.children(':first').height());
              divHolder = iElement.children(':first').children(':last');
              iElement.children(':first').children(':last').remove();
              iElement.children(':first').next().next().prepend(divHolder);
            }
          }
          $timeout((function() {
            if (iElement.children(':first').next().next().height() > scope.maxColHeight) {
              while (iElement.children(':first').next().next().height() > scope.maxColHeight) {
                log.debug(iElement.children(':first').next().next().height());
                divHolder = iElement.children(':first').next().next().children(':last');
                iElement.children(':first').next().next().children(':last').remove();
                iElement.children(':first').next().next().next().next().prepend(divHolder);
              }
            }
          }), 1);
        };
      }
    };
  }
]);
