    This Module is to hold the Phase info and to construct the expand and collapse tree in Phase level

    angular.module('calview.directives').directive('accordionPhase', [
  '$timeout', '$compile', function($timeout, $compile) {
    return {
      templateUrl: "accordionphasedir.html",
      replace: true,
      restrict: 'E',
      scope: {
        array: '=',
        index: '@'
      },
      link: function(scope, iElement, iAttrs, controller) {
        var watchers;
        scope.phase = scope.array[scope.index];
        scope.backColor = iElement.css('background-color');
        scope.uids = scope.phase.uids();
        scope.noResults = i18n.oncology.calview.NO_RESULTS;
        scope.hiddenMsg = i18n.oncology.calview.NO_ITEMS_FOUND;
        scope.ordersLabel = i18n.oncology.calview.ORDERS;
        scope.resultsHidden = false;
        scope.isPhaseExpanded = true;
        scope.statusWidth = 0;
        scope.areOrdersExpanded = true;
        scope.$watch('areOrdersExpanded', function(areOrdersExpanded) {
          if (areOrdersExpanded) {
            scope.$broadcast('orderTextWidth');
          }
        });
        scope.$on('phaseTextWidth', function() {
          $timeout((function() {
            var $chemoStatus;
            $chemoStatus = iElement.find('.chemo-status');
            scope.statusWidth = $chemoStatus.width();
          }), 100);
        });

        watchers = void 0;
        scope.$on('suspendWatchers', function() {
          watchers = scope.$$watchers;
          scope.$$watchers = [];
        });
        scope.$on("resumeWatchers", function() {
          if (watchers) {
            scope.$$watchers = watchers;
          }
          watchers = void 0;
        });
        scope.dates = [];
      }
    };
  }
]);
