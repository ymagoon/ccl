    angular.module('calview.directives').directive('accordionChemo', [
  '$timeout', '$compile', function($timeout, $compile) {
    return {
      template: "<div class=\"row-fluid\" data-ng-show=\"!resultsHidden\" data-ng-class=\"{'slidedown-animation': !resultsHidden}\"><div class=\"accordion-line\" data-ng-if=\"chemo.regimenId() != prevRegimenId && chemo.regimenId() != ''\"><strong>{{chemo.regimenName()}}</strong></div><div class=\"accordion-line accordion-text chemo-line clickable\"><div class=\"chemo-status\">{{chemo.status()}}</div><div class=\"text-overflow\" data-ng-style=\"{'margin-right': statusWidth }\" data-ng-click=\"isChemoExpanded = !isChemoExpanded;\"><a class=\"anchor accordion-toggle\" href=\"#\"><i data-ng-class=\"{'icon-play': !isChemoExpanded, 'opened': isChemoExpanded}\"></i>\n        {{chemo.name()}}\n      </a></div></div><div data-ng-show=\"isChemoExpanded\" data-ng-class=\"{'slidedown-animation': isChemoExpanded}\"><div data-ng-show=\"arePhasesExpanded\" data-ng-class=\"{'slidedown-animation': arePhasesExpanded}\"><accordion-phase class=\"plan-sublist\" data-array=\"chemo.phases\" data-index=\"{{$index}}\" data-ng-repeat=\"phase in chemo.phases track by phase.uid\"></accordion-phase></div></div></div>",
      replace: true,
      restrict: 'E',
      scope: {
        array: '=',
        index: '@'
      },
      link: function(scope, iElement, iAttrs, controller) {
        var $firstBorn, watchers;
        scope.noResults = i18n.oncology.calview.NO_RESULTS;
        scope.phasesLabel = i18n.oncology.calview.PHASES;
        scope.hiddenMsg = i18n.oncology.calview.NO_ITEMS_FOUND;
        scope.chemo = scope.array[scope.index];
        if (scope.array[scope.index - 1] != null) {
          scope.prevRegimenId = scope.array[scope.index - 1].regimenId();
        }
        scope.backColor = iElement.css('background-color');
        scope.uids = scope.chemo.uids();
        scope.isChemoExpanded = true;
        scope.arePhasesExpanded = true;
        scope.statusWidth = 0;
        scope.resultsHidden = false;
        scope.showHidden = function() {
          scope.resultsHidden = scope.chemo.ordersHidden();
        };
        scope.$watch('arePhasesExpanded', function(arePhasesExpanded) {
          if (arePhasesExpanded) {
            scope.$broadcast('phaseTextWidth');
          }
        });
        scope.$on('chemoOpen', function() {
          $timeout((function() {
            var $chemoStatus;
            $chemoStatus = iElement.find('.chemo-status');
            scope.statusWidth = $chemoStatus.width();
          }), 100);
        });
        scope.$on('showEst', function(ev, showEst) {
          scope.showHidden();
        });
        scope.$on('showSched', function(ev, showSched) {
          scope.showHidden();
        });
        watchers = void 0;
        scope.$on('suspendWatchers', function() {
          watchers = scope.$$watchers;
          scope.$$watchers = [];
        });
        scope.$on("resumeWatchers", function() {
          if (watchers) {
            scope.$$watchers = watchers;
          }
          watchers = void 0;
        });
        scope.popdata = {
          content: {
            data: scope.chemo.popupData(),
            title: scope.chemo.popupTitle()
          }
        };
        $firstBorn = iElement.children(':first');
        $firstBorn.attr({
          'data-highlight': 'uids',
          'data-qtip': 'popdata',
          'data-ng-click': 'isChemoExpanded = !isChemoExpanded'
        });
        $compile($firstBorn)(scope);
      }
    };
  }
]);
