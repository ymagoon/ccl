    This module is useful to construct the backward, forward, filters and Print options in calendar view

    angular.module('calview.directives').directive('calendarControls', [
  'pubsub', function(pubsub) {
    return {
      templateUrl: "calendarcontrolsdir.html",
      replace: true,
      restrict: 'E',
      scope: {
        movebackward: '&',
        moveforward: '&',
        startindex: '='
      },
      link: function(scope, iElement, iAttrs, controller) {
        var backFive, backOne, forwardFive, forwardOne, watchers;
        backFive = i18n.oncology.calview.BACK_FIVE;
        backOne = i18n.oncology.calview.BACK_ONE_W;
        forwardOne = i18n.oncology.calview.FORWARD_ONE_W;
        forwardFive = i18n.oncology.calview.FORWARD_FIVE;
        scope.addNote = i18n.oncology.calview.ADD_NOTE;
        scope.fiveWeeks = i18n.oncology.calview.FIVEWEEKS;
        scope.backFive = backFive;
        scope.backOne = backOne;
        scope.forwardOne = forwardOne;
        scope.forwardFive = forwardFive;
        scope.endIndex = vars.daysRange;
        scope.printPreview = i18n.oncology.calview.PRINT_PREVIEW;
        scope.scheduleEvent = i18n.oncology.calview.SCHED_EVENT;
        scope.dropDownImg = vars.staticContentLocation + '/img/6361_16.gif';
        scope.$on('showAccord', function(ev, toggleValue) {
          if (toggleValue) {
            scope.accordTitle = i18n.oncology.calview.HIDE_CAL_ITEMS;
          } else {
            scope.accordTitle = i18n.oncology.calview.SHOW_CAL_ITEMS;
          }
        });
        scope.toggleAccordTitle = function() {
          scope.$parent.showAccord = !scope.$parent.showAccord;
        };
        scope.$on('showList', function(ev, toggleValue) {
          if (toggleValue) {
            scope.listTitle = i18n.oncology.calview.HIDE_CAL_LIST;
          } else {
            scope.listTitle = i18n.oncology.calview.SHOW_CAL_LIST;
          }
        });
        scope.toggleListTitle = function() {
          scope.$parent.showList = !scope.$parent.showList;
        };
        scope.$on('showEst', function(ev, toggleValue) {
          if (toggleValue) {
            scope.estTitle = i18n.oncology.calview.HIDE_EST_EVENTS;
          } else {
            scope.estTitle = i18n.oncology.calview.SHOW_EST_EVENTS;
          }
        });
        scope.toggleEstTitle = function() {
          scope.$parent.showEst = !scope.$parent.showEst;
        };
        scope.$on('showSched', function(ev, toggleValue) {
          if (toggleValue) {
            scope.schedTitle = i18n.oncology.calview.HIDE_SCHED_EVENTS;
          } else {
            scope.schedTitle = i18n.oncology.calview.SHOW_SCHED_EVENTS;
          }
        });
        scope.toggleSchedTitle = function() {
          scope.$parent.showSched = !scope.$parent.showSched;
        };
        scope.apptsLabel = i18n.oncology.calview.APPTS.toUpperCase();
        scope.$on('showAppts', function(ev, toggleValue) {
          if (toggleValue) {
            scope.apptsTitle = i18n.oncology.calview.HIDE_APPTS;
          } else {
            scope.apptsTitle = i18n.oncology.calview.SHOW_APPTS;
          }
        });
        scope.toggleApptsTitle = function() {
          scope.$parent.showAppts = !scope.$parent.showAppts;
        };
        scope.chemoLabel = i18n.oncology.calview.CHEMO.toUpperCase();
        scope.$on('showChemo', function(ev, toggleValue) {
          if (toggleValue) {
            scope.chemoTitle = i18n.oncology.calview.HIDE_CHEMO;
          } else {
            scope.chemoTitle = i18n.oncology.calview.SHOW_CHEMO;
          }
        });
        scope.toggleChemo = function() {
          scope.$parent.showChemo = !scope.$parent.showChemo;
        };
        scope.medsLabel = i18n.oncology.calview.RX;
        scope.$on('showMeds', function(ev, toggleValue) {
          if (toggleValue) {
            scope.medsTitle = i18n.oncology.calview.HIDE_MEDS;
          } else {
            scope.medsTitle = i18n.oncology.calview.SHOW_MEDS;
          }
        });
        scope.toggleMedsTitle = function() {
          scope.$parent.showMeds = !scope.$parent.showMeds;
        };
        scope.$on('showNotes', function(ev, toggleValue) {
          if (toggleValue) {
            scope.notesTitle = i18n.oncology.calview.HIDE_NOTES;
          } else {
            scope.notesTitle = i18n.oncology.calview.SHOW_NOTES;
          }
        });
        scope.toggleNotesTitle = function() {
          scope.$parent.showNotes = !scope.$parent.showNotes;
        };
        watchers = void 0;
        scope.$on('suspendWatchers', function() {
          watchers = scope.$$watchers;
          scope.$$watchers = [];
        });
        scope.$on("resumeWatchers", function() {
          if (watchers) {
            scope.$$watchers = watchers;
          }
          watchers = void 0;
        });
        scope.showPrintModal = function() {
          scope.$emit('showPrintModal');
        };
      }
    };
  }
