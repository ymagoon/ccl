    angular.module('calview.services').factory('notesMedsSvc', [
  'dao', function(dao) {
    return {
      addNotesMeds: function(orderId, calItemsId, noteText, typeFlag, dates, cb) {
        var createNewItem, e, modifyExistingItem;
        createNewItem = function() {
          return {
            error: function(data, status, headers, config) {
              log.error("status: " + status + "\n");
              log.error("data: " + data + "\n\n");
              $('#loading-spinner').hide();
              alert('Call to onc_tc_add_cal_items failed.  The note or prescription has not been added.');
              cb(void 0);
            },
            params: "'MINE'," + orderId + ".0,\"" + noteText + "\"," + typeFlag + ",'" + dates + "'," + vars.person_id + ".0",
            success: function(data, status, headers, config) {
              var json;
              log.debug("status: " + status + "\n");
              log.debug("data: " + (JSON.stringify(data[0])) + "\n\n");
              json = data[0].RECORD_DATA;
              cb(json.ONC_TC_CAL_ITEMS_ID);
               $('#loading-spinner').hide();
            },
            url: 'onc_tc_add_cal_items'
          };
        };
        modifyExistingItem = function() {
          return {
            error: function(data, status, headers, config) {
              log.error("status: " + status + "\n");
              log.error("data: " + data + "\n\n");
               $('#loading-spinner').hide();
              alert('Call to onc_tc_updt_cal_items failed.  The note or prescription has not been updated.');
              cb(void 0);
            },
            params: "'MINE'," + orderId + ".0,\"" + noteText + "\"," + typeFlag + ",'" + dates + "'," + calItemsId + ".0",
            success: function(data, status, headers, config) {
              log.debug("status: " + status + "\n");
              log.debug("data: " + (JSON.stringify(data[0])) + "\n\n");
              cb(calItemsId);
              $('#loading-spinner').hide();
            },
            url: 'onc_tc_updt_cal_items'
          };
        };
        try {
          $('#loading-spinner').show();
          if (calItemsId === 0) {
            dao.call(createNewItem());
          } else {
            dao.call(modifyExistingItem());
          }
        } catch (_error) {
          e = _error;
           $('#loading-spinner').hide();
          return log.error(e);
        }
      },
      deleteItem: function(calItemsId, cb) {
