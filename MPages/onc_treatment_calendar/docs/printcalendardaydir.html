    This module is useful to construct the days cells information in print calendar view

    var DOTW;

DOTW = {
  'SUN': 0,
  'MON': 1,
  'TUE': 2,
  'WED': 3,
  'THU': 4,
  'FRI': 5,
  'SAT': 6
};

angular.module('calview.directives').directive('printCalendarDay', function() {
  return {
    templateUrl: "printcalendardaydir.html",
    replace: true,
    restrict: 'E',
    scope: true,
    link: function(scope, iElement, iAttrs, controller) {
    	scope.day = moment().day(DOTW[iAttrs.dotw]).format('ddd');
    }
  };
});
