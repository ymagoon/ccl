    ListCtrl controller is useful to hold the List view in treatment Calendar it will enable scroll bar based on hieght of the screen

    var mod;

mod = angular.module('calview.controllers').controller('ListCtrl', [
  '$scope', function($scope) {
    $scope.calList = i18n.oncology.calview.CALENDAR_LIST;
    $scope.showAll = true;
    $scope.showAllMsg = i18n.oncology.calview.ALL_DAYS;
    $scope.$watch('showAll', function(showAll) {
      $scope.$broadcast('showAllDays', showAll);
    });
    $scope.month = _.str.capitalize(moment().format('MMMM YYYY'));
    $scope.$on('setListMonth', function(ev, month) {
      $scope.month = month;
    });
  }
]);

mod.config(function() {
  var $el, parHt;
  $el = $('#cal4');
  parHt = $el.parent().height();
  $el.height(parHt);
  $el.children(':first').height(parHt-16);
});
