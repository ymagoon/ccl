    angular.module('calview.directives').directive 'displayFilters', ->
  template:
    """
    <header class="row-fluid">
      <div class="btn-group">
        <button type="button" class="btn btn-filters" data-toggle="dropdown"></button>
        <ul class="dropdown-menu" role="menu">
          <li><a data-ng-click="toggleAccordTitle()">{{accordTitle}}</a></li>
          <li><a data-ng-click="toggleListTitle()">{{listTitle}}</a></li>
          <li><a data-ng-click="toggleEstTitle()">{{estTitle}}</a></li>
          <li><a data-ng-click="toggleSchedTitle()">{{schedTitle}}</a></li>
          <li><a data-ng-click="toggleApptsTitle()">{{apptsTitle}}</a></li>
          <li><a data-ng-click="toggleChemo()">{{chemoTitle}}</button></li>
          <li><a data-ng-click="toggleMedsTitle()">{{medsTitle}}</a></li>
        </ul>
      </div>
    </header>
    """
  replace: true
  restrict: 'E'
  scope: true
  link: (scope, iElement, iAttrs, controller) ->
    scope.dropDownImg = vars.staticContentLocation + '/img/6361_16.png'
    hideAccordTitle = i18n.oncology.calview.HIDE_CAL_ITEMS
    showAccordTitle = i18n.oncology.calview.SHOW_CAL_ITEMS
    scope.accordTitle = hideAccordTitle
    scope.toggleAccordTitle = ->
      scope.$parent.showAccord = not scope.$parent.showAccord
      scope.accordTitle = if scope.$parent.showAccord then hideAccordTitle else showAccordTitle
      return
    hideListTitle = i18n.oncology.calview.HIDE_CAL_LIST
    showListTitle = i18n.oncology.calview.SHOW_CAL_LIST
    scope.listTitle = hideListTitle
    scope.toggleListTitle = ->
      scope.$parent.showList = not scope.$parent.showList
      scope.listTitle = if scope.$parent.showList then hideListTitle else showListTitle
      return
    hideEstTitle = i18n.oncology.calview.HIDE_EST_EVENTS
    showEstTitle = i18n.oncology.calview.SHOW_EST_EVENTS
    scope.estTitle = hideEstTitle
    scope.toggleEstTitle = ->
      scope.$parent.showEst = not scope.$parent.showEst
      scope.estTitle = if scope.$parent.showEst then hideEstTitle else showEstTitle
      return
    hideSchedTitle = i18n.oncology.calview.HIDE_SCHED_EVENTS
    showSchedTitle = i18n.oncology.calview.SHOW_SCHED_EVENTS
    scope.schedTitle = hideSchedTitle
    scope.toggleSchedTitle = ->
      scope.$parent.showSched = not scope.$parent.showSched
      scope.schedTitle = if scope.$parent.showSched then hideSchedTitle else showSchedTitle
      return
    hideApptsTitle = i18n.oncology.calview.HIDE_APPTS
    showApptsTitle = i18n.oncology.calview.SHOW_APPTS
    scope.apptsLabel = i18n.oncology.calview.APPTS.toUpperCase()
    scope.apptsTitle = hideApptsTitle
    scope.toggleApptsTitle = ->
      scope.$parent.showAppts = not scope.$parent.showAppts
      scope.apptsTitle = if scope.$parent.showAppts then hideApptsTitle else showApptsTitle
      return
    hideChemoTitle = i18n.oncology.calview.HIDE_CHEMO
    showChemoTitle = i18n.oncology.calview.SHOW_CHEMO
    scope.chemoLabel = i18n.oncology.calview.CHEMO.toUpperCase()
    scope.chemoTitle = hideChemoTitle
    scope.toggleChemo = ->
      scope.$parent.showChemo = not scope.$parent.showChemo
      scope.chemoTitle = if scope.$parent.showChemo then hideChemoTitle else showChemoTitle
      return
