app.get('/onc_tc_get_sched_items', function(req, res) {
  MongoClient.connect("mongodb://" + ip.address + ":27017/calview", function(err, db) {
    var orderables;
    if (err) {
      db.close();
      throw err;
    }
    orderables = db.collection('orderables');
    orderables.find().toArray(function(err, arr) {
      if (err) {
        db.close();
        throw err;
      }
      res.send(arr);
      db.close();
    });
  });
});
