    this Module is to hold the appointment info along with plan and regimen data.

    angular.module('calview.directives').directive('accordionOrder', [
  'pubsub', '$compile', function(pubsub, $compile) {
    return {
      templateUrl: "accordionorderdir.html",
      replace: true,
      restrict: 'E',
      scope: {
        array: '=',
        index: '@'
      },
      link: function(scope, iElement, iAttrs, controller) {
        var $firstBorn, $lastBorn, watchers;
        scope.order = scope.array[scope.index];
        scope.backColor = iElement.css('background-color');
        scope.uids = scope.order.uids();
        scope.date = "" + (scope.order.isEst() ? '*' : '') + (scope.order.date());

        scope.$watch('order.json.ORDER_DATE', function(newDate, oldDate) {
          scope.date = "" + (scope.order.isEst() ? '*' : '') + (scope.order.date());
          
          scope.popdata = {
          content: {
            data: scope.order.popupData(),
          }
        };

        $firstBorn.attr({
          'data-qtip': 'popdata'
        });
        $lastBorn.attr({
          'data-qtip': 'popdata'
        });
        $compile($firstBorn)(scope);
        $compile($lastBorn)(scope);
          
        });
        watchers = void 0;
        scope.$on('suspendWatchers', function() {
          watchers = scope.$$watchers;
          scope.$$watchers = [];
        });
        scope.$on("resumeWatchers", function() {
          if (watchers) {
            scope.$$watchers = watchers;
          }
          watchers = void 0;
        });
        scope.showModal = function() {
          scope.$emit('showModal', false, false, scope.order.name(), scope.order.toUnix(), scope.order.uid, scope.order);
        };
        scope.popdata = {
          content: {
            data: scope.order.popupData()
          }
        };
        $firstBorn = iElement.children(':first');
        $lastBorn = iElement.children(':last');
        $firstBorn.attr({
          'data-qtip': 'popdata'
        });
        $lastBorn.attr({
          'data-qtip': 'popdata'
        });
        $compile($firstBorn)(scope);
        $compile($lastBorn)(scope);
      }
    };
  }
]);
