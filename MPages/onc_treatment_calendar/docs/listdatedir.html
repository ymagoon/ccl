    This module to construct the date row entry in list view

    angular.module('calview.directives').directive('listDate', [
  'pubsub', function(pubsub) {
    return {
      templateUrl: "listdatedir.html",
      replace: true,
      restrict: 'E',
      scope: {
        array: '=',
        index: '@',
        showAll: '@'
      },
      link: function(scope, iElement, iAttrs, controller) {
        var $firstBorn, $lastBorn, mom, showDate, watchers,todayMidnight;         
        todayMidnight = moment().startOf('day');
        showDate = function(show) {
          scope.showDate = show || !(scope.date_.noAppts() && scope.date_.noChemo() && scope.date_.noMeds() && scope.date_.noNotes());
        };
        scope.currentIndex = scope.index;
        scope.date_ = scope.array[scope.index];
        scope.month = _.str.capitalize(moment(scope.date_.unixStamp).format('MMMM YYYY'));
        showDate(/true/.test(scope.showAll));
        scope.mouseOver = false;
        scope.addNote = i18n.oncology.calview.ADD_NOTE;
        scope.uids = scope.date_.uids();
        
        mom = moment(scope.date_.unixStamp);
        scope.day = mom.format('dd').charAt(0);
        scope.date = mom.date();
		scope.isSaturday = function() {
          return mom.day() === 6;
        };
        scope.$on('showAllDays', function(ev, showAll) {
          showDate(showAll);
        });
       
        scope.$on('scrolling', function(ev, offset) {
          var myTop, _ref;
          myTop = $(iElement).offset().top;
          if ((0 <= (_ref = Math.abs(myTop - offset)) && _ref < 3)) {
            scope.$emit('setListMonth', _.str.capitalize(mom.format('MMMM YYYY')));
          }
        });
        
        watchers = void 0;
        scope.$on('suspendWatchers', function() {
          watchers = scope.$$watchers;
          scope.$$watchers = [];
        });
        scope.$on("resumeWatchers", function() {
          if (watchers) {
            scope.$$watchers = watchers;
          }
          watchers = void 0;
        });
         scope.isPast = function() {
          return mom.diff(todayMidnight) < 0;
        };
        scope.showModal = function($index) {
          var sched, schedType;
          if ($index >= 0) {
            sched = scope.date_.schedulables[$index];
            schedType = sched.constructor.toString().match(/^function\s*([^\s(]+)/)[1];
            if (sched.isMulti === true && sched.isNote === false) {
              if (sched.oneTimeInd() === 0 && sched.allDates() === '') {
                scope.$emit('showModal', sched.isMulti, sched.isNote, sched.text(), sched.toUnix(), sched.uid, sched);
              }
            } else {
              scope.$emit('showModal', sched.isMulti, sched.isNote, sched.text(), sched.toUnix(), sched.uid, sched);
            }
          } else {
            scope.$emit('showModal', true, true, '', scope.date_.unixStamp, void 0, void 0);
          }
        };
      }
    };
  }
]);
